function preOrder(root) {
  if (!root) {
    return
  }
  console.log(root.val)
  preOrder(root.left)
  preOrder(root.right)
}

function inOrder(root) {
  if (!root) {
    return
  }
  inOrder(root.left)
  console.log(root.val)
  inOrder(root.right)
}

function afterOrder(root) {
  if (!root) {
    return
  }
  afterOrder(root.left)
  afterOrder(root.right)
  console.log(root.val)
}
